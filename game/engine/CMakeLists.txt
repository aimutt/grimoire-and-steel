# Bake character portraits + the startup splash into the executable as Windows RCDATA
# resources (rather than shipping loose files). At configure time we scan the tracked asset
# folder and generate a .rc plus a small header mapping portrait filename -> resource id.
file(GLOB ENGINE_PORTRAITS CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/assets/portraits/*.png")
list(SORT ENGINE_PORTRAITS)

set(_rc "// generated by CMake -- do not edit\n")
set(_hdr "// generated by CMake -- do not edit\n#pragma once\n#include <string>\n#include <vector>\n")
string(APPEND _hdr "struct EmbeddedPortrait { const char* file; const char* res; };\n")
string(APPEND _hdr "inline const std::vector<EmbeddedPortrait> kEmbeddedPortraits = {\n")
foreach(png ${ENGINE_PORTRAITS})
    get_filename_component(_fname "${png}" NAME)     # portrait01.png
    get_filename_component(_stem "${png}" NAME_WE)   # portrait01
    string(TOUPPER "${_stem}" _id)                   # PORTRAIT01
    string(REPLACE "/" "\\\\" _rcpath "${png}")      # backslash-escaped abs path for .rc
    string(APPEND _rc "${_id} RCDATA \"${_rcpath}\"\n")
    string(APPEND _hdr "    { \"${_fname}\", \"${_id}\" },\n")
endforeach()
string(APPEND _hdr "};\n")

set(_splash "${CMAKE_CURRENT_SOURCE_DIR}/assets/gns-splash.png")
string(REPLACE "/" "\\\\" _splashrc "${_splash}")
string(APPEND _rc "GNSSPLASH RCDATA \"${_splashrc}\"\n")

file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/embedded_assets.rc" "${_rc}")
file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/embedded_assets.h" "${_hdr}")

# Shared shop item-art catalog (baked in; same images the Module Creator offers).
gns_generate_item_rc(
    "${CMAKE_CURRENT_BINARY_DIR}/embedded_items.rc"
    "${CMAKE_CURRENT_BINARY_DIR}/embedded_items.h")

add_executable(GameEngine src/main.cpp
    "${CMAKE_CURRENT_BINARY_DIR}/embedded_assets.rc"
    "${CMAKE_CURRENT_BINARY_DIR}/embedded_items.rc")
target_include_directories(GameEngine PRIVATE "${CMAKE_CURRENT_BINARY_DIR}")
target_link_libraries(GameEngine PRIVATE
    gns_core gns_ui imgui SDL2::SDL2 SDL2::SDL2main SDL2_image::SDL2_image)
gns_copy_runtime(GameEngine)
